function OutputInt: 
	PARAMETER vArr[0]
BLOCK
END OutputInt


function fib: 
	PARAMETER vArr[4]
	vArr[8] = 0
	vArr[12] = 1
	vArr[16] = 0
	vArr[20] = 0
	WHILE_1:
	t0 = vArr[20] < vArr[4]
	LOOP_COND t0 < vArr[4] GOTO END_WHILE_1
		PARAM vArr[8]
		CALL OutputInt, 1
		t1 = vArr[8] + vArr[12]
		vArr[16] = t1
		vArr[8] = vArr[12]
		vArr[12] = vArr[16]
		t2 = vArr[20] + 1
		vArr[20] = t2
	BLOCK
	GOTO WHILE_1
	END_WHILE_1:
BLOCK
END fib


function main: 
	vArr[33] = 7
	PARAM vArr[33]
	CALL fib, 1
BLOCK
END main
